# 🔹 Subtask 1.4: Create the Jinja2 Template for httpd.conf

## 🛠️ Create Apache Configuration Template

```bash
cat > templates/httpd.conf.j2 << 'EOF'
#
# Apache HTTP Server Configuration File
# Generated by Ansible Template on {{ ansible_date_time.iso8601 }}
# Managed by: {{ ansible_user }}@{{ ansible_hostname }}
#

ServerRoot "/etc/httpd"
Listen {{ apache_listen_port }}

{% if enable_ssl %}
Listen {{ ssl_port }} ssl
{% endif %}

{% if custom_modules is defined %}
{% for module in custom_modules %}
LoadModule {{ module }}_module modules/{{ module }}.so
{% endfor %}
{% endif %}

User apache
Group apache

ServerAdmin {{ apache_server_admin }}
ServerName {{ apache_server_name }}:{{ apache_port }}

DocumentRoot "{{ apache_document_root }}"

<Directory "{{ apache_document_root }}">
    Options Indexes FollowSymLinks
    AllowOverride None
    Require all granted
</Directory>

{% if app_environment == 'production' %}
MaxRequestWorkers {{ apache_max_clients }}
ThreadsPerChild {{ apache_threads_per_child }}
Timeout {{ apache_timeout }}
KeepAlive On
{% else %}
MaxRequestWorkers {{ apache_max_clients }}
ThreadsPerChild 10
Timeout 60
KeepAlive Off
{% endif %}

ErrorLog "logs/error_log"
LogLevel {{ apache_log_level }}

IncludeOptional conf.d/*.conf
EOF
```

# 🧪 Task 2: Apply the Template using ansible-playbook

---

# 🔹 Subtask 2.1: Create Main Playbook

## 🛠️ Create Deployment Playbook

```bash
cat > deploy-apache-templates.yml << 'EOF'
---
- name: Deploy Apache Web Server with Templates
  hosts: webservers
  become: yes
  gather_facts: yes

  tasks:

    - name: Install Apache
      yum:
        name: httpd
        state: present

    - name: Deploy Apache Config
      template:
        src: httpd.conf.j2
        dest: /etc/httpd/conf/httpd.conf
        owner: root
        group: root
        mode: '0644'
      notify: restart apache

    - name: Deploy Website
      template:
        src: index.html.j2
        dest: "{{ apache_document_root }}/index.html"
        owner: apache
        group: apache
        mode: '0644'

    - name: Start Apache
      systemd:
        name: httpd
        state: started
        enabled: yes

  handlers:
    - name: restart apache
      systemd:
        name: httpd
        state: restarted
EOF
```

# 🔹 Subtask 3.2: Deploy System Information Template

## 🛠️ Create Deployment Playbook

```bash
cat > deploy-system-info.yml << 'EOF'
---
- name: Deploy System Information Dashboard
  hosts: webservers
  become: yes
  gather_facts: yes

  tasks:

    - name: Create Directory
      file:
        path: "{{ apache_document_root }}/system-info"
        state: directory
        owner: apache
        group: apache
        mode: '0755'

    - name: Deploy Template
      template:
        src: system-info.html.j2
        dest: "{{ apache_document_root }}/system-info/index.html"
        owner: apache
        group: apache
        mode: '0644'
EOF
```

---

## ▶️ Run the Playbook

```bash
ansible-playbook -i inventory deploy-system-info.yml -v
```

---

# 🔹 Subtask 3.3: Create Virtual Host Template

## 🛠️ Create vhost.conf.j2

```bash
cat > templates/vhost.conf.j2 << 'EOF'
<VirtualHost *:{{ apache_listen_port }}>

    ServerName {{ ansible_fqdn }}
    DocumentRoot {{ apache_document_root }}

    {% if app_environment == 'production' %}
    LogLevel error
    {% else %}
    LogLevel debug
    {% endif %}

    ErrorLog logs/{{ ansible_hostname }}_error.log
    CustomLog logs/{{ ansible_hostname }}_access.log combined

</VirtualHost>
EOF
```

---

# 🔹 Subtask 3.4: Deploy Virtual Host

## 🛠️ Create Playbook

```bash
cat > deploy-vhost.yml << 'EOF'
---
- name: Deploy Virtual Host
  hosts: webservers
  become: yes

  tasks:

    - name: Deploy VHost Config
      template:
        src: vhost.conf.j2
        dest: /etc/httpd/conf.d/{{ ansible_hostname }}.conf
        owner: root
        group: root
        mode: '0644'

    - name: Reload Apache
      systemd:
        name: httpd
        state: reloaded
EOF
```

---

## ▶️ Run Playbook

```bash
ansible-playbook -i inventory deploy-vhost.yml -v
```
